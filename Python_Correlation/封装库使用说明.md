## Python 功能库封装调用说明



## logger_manager.py

logger_manager.py说明：是对 python  logging 库进行封装，将使用方式简单化

第一种调用方式，也是最简单的使用方式：

```python
#!/usr/bin/env python
# -*-coding:utf-8 -*-

# 调用不传任何参数，使用封装好的默认参数
from logger_manager import logger

class TestDemo:
    def __init__(self):
        self.test_function()

    def test_function(self):
        # 默认日志也会输出在终端上，同时会创建日志文件，保存在项目的根目录中（默认会创建normal.log、error.log）
        logger.debug("This is a debug message.")
        logger.info("This is a test message.")
        logger.warning("This is a warning message.")
        logger.error("This is an error message.")
        logger.critical("This is a critical message.")

if __name__ == '__main__':
    test = TestDemo()
```



第二种调用方式，取消单例模式：

```python
#!/usr/bin/env python
# -*-coding:utf-8 -*-

# logger = CustomLogger()   # ❌ 删除或注释这一行

from logger_manager import CustomLogger

class TestDemo:
    def __init__(self):
        # 取消了库中的单例模式，可以进行传入参数，根据参数来输出或保存日志
        self.logger = CustomLogger(
            log_folder='D:\\Temp\\TestLog',
            log_level=10, # logging.DEBUG
            max_bytes=20 * 1024 * 1024,
            backup_count=3,
            enable_color=False,
            enable_save_log=True,
            enable_terminal_print=True,
            normal_log_name='app.log',
            error_log_name='app_error.log'
        )
        self.test_function()

    def test_function(self):
        # 根据传入的参数进行变化
        self.logger.debug('This is a debug message')
        self.logger.info('This is a test message')
        self.logger.warning('This is a warning message')
        self.logger.error('This is an error message')
        self.logger.critical('This is a critical message')

if __name__ == '__main__':
    test = TestDemo()
```



## mqtt_manager.py

mqtt_manager.py说明：是对 python paho.mqtt.client 库进行封装，方便其他文件调用简单化

调用示例：

```python
#!/usr/bin/env python
# -*-coding:utf-8 -*-

import time
from mqtt_manager import EnhancedMQTTClient

class TestDemo:
    def __init__(self):
        # 初始化客户端（具体传参可查看封装库）
        self.client = EnhancedMQTTClient(
            host="broker.emqx.io",
            port=8083,
            scheme="ws",
            # username="admin",
            # password="admin",
            auto_reconnect=True,
            reconnect_attempts=3
        )
        self.test_function()

    def test_function(self):
        # 先注册处理器（此时未连接，不会立即订阅，不推荐此方法）
        # client.add_handler("sensors/temperature", temperature_handler, qos=1)

        try:
            # 启用TLS
            # client.enable_tls(ca_certs="broker.emqx.io-ca.crt")

            # 建立连接（连接成功后会自动订阅已注册的主题）
            self.client.connect()
            print(f"[INFO] MQTT连接状态: {self.client._connect_status.get()}")

            # 订阅消息
            self.client.add_handler("device/status", self.device_handler, qos=0)
            self.client.add_handler("sensors/temperature", self.temperature_handler, qos=1)
            self.client.add_handler("system/status", self.status_handler, qos=0)

            # 模拟程序耗时
            time.sleep(2)

            # 发布测试消息
            self.client.publish("sensors/temperature", {"value": 25.6, "unit": "℃"})
            self.client.publish("system/status", "operational")
            self.client.publish("device/status", "online")

            # 保持连接并处理消息
            while True:
                self.client.publish("sensors/temperature", {"value": 33.3, "unit": "℃"})
                time.sleep(1)

        except KeyboardInterrupt:
            self.client.disconnect()
        except Exception as e:
            print(f"[ERROR] 发生错误: {str(e)}")
            self.client.disconnect()

    def temperature_handler(self, topic: str, data: dict):
        print(f"温度数据更新: {data['value']}℃")

    def status_handler(self, topic: str, message: str):
        print(f"系统状态: {message}")

    def device_handler(self, topic: str, message: str):
        print(f"设备状态: {message}")

if __name__ == '__main__':
    test = TestDemo()
```



## ups_info.py

ups_info.py说明：用与查询网络UPS状态

调用使用示例：

```python
#!/usr/bin/env python
# -*-coding:utf-8 -*-

import time
from ups_info import UPSInfo

class TestDemo:
    def __init__(self):
        # 初始化
        self.ups_info = UPSInfo(
            nut_server="192.168.100.220",
            nut_port=3493,
            ups_name="ups")

        self.test_function()

    def test_function(self):
        print(self.ups_info.get_ups_data())

if __name__ == '__main__':
    test = TestDemo()
```
